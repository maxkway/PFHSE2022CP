def cartesian_to_polar(c):
    """ ,   10   xy 
       r phi.
    
    Parameters
    ----------
    c: Array of int.
      102,   --   . 
    
    Returns
    -------
    p: Array of int. 
         .    r.
    """#      10x2      
    p = np.empty((10, 2), dtype=int)
    p[:, 1] = np.arctan((c[:, 1]/c[:, 0]))
    p[:, 0] = np.sqrt((c[:, 0])**2 + (c[:, 1])**2)

def make_curve(n, a):
    '''   n2,  n  
        '''
    phi = np.linspace(1, 10, num=n)
    r = a + np.cos(phi)
    xy = np.empty((n, 2))
    xy[:, 0] = r * np.cos(phi)
    xy[:, 1] = r * np.sin(phi)
    #raise NotImplementedError()
    return xy
