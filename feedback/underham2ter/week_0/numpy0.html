def make_curve(n, a):
    '''   n2,  n  
        '''
    phi = np.linspace(0, 2*np.pi, n)
    p = [[a+np.cos(i), i] for i in phi]
    xy = [[r * np.cos(phi), r * np.sin(phi)] for r, phi in p]
    return xy
