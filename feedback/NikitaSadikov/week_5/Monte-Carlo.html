def getSphereVolumeExact(D, R = 1.0):
    """
        D-   R   

    --------
    
    D - int,   
    R = 1 - float,  
    --------
     :
    V - float,  
    """
    V = 1
    for k in range(D):
        g = (2 + k) / 2
        V = V * np.pi**0.5 * R * sp.gamma(g)/sp.gamma(g+1/2)
    return V

def getSphereVolumePseudorandom(N, D, R = 1):
    """
        D-   R  -   

    --------
    :
    N - int,   
    D - int,   
    R = 1 - float,  
    --------
     :
    V - float,  
    """

    p = np.random.uniform(low = -R, high = R, size = (N, D))
    in_1 = np.count_nonzero([np.linalg.norm(p[i]) < R for i in range(N)])
    V= (2 * R) ** D
    V = in_1 / N * V

    return V

def getSample(N):
    w = lambda x: np.sqrt(1 - x ** 2)
    s = []
    c = 0
    while c < N:
        x = np.random.uniform(-1,1)
        y = np.random.uniform(0,1)
        if y < w(x):
            s.append(x)
            c += 1
    return s

def makeFlip(T, state):
    '''
     N^2   ,        ,   4.  5.
    
    ---------
    :
    T - float,  ,  ,  ,  kT/J
    state - numpy ndarray of ints,     NxN
    --------
     :
    state - numpy ndarray of ints,      NxN
    '''
    n = state.shape[0]
    for c in range(N * N):
        i = np.random.randint(0, high=n)
        j = np.random.randint(0, high=n)
        d = getDeltaE(i, j, state)
        if d < 0:
            state[i,j] *= -1
        else:
            if np.random.sample() < np.exp(-d / T):
                state[i,j] *= -1
    return state

def getEnergy(state):
    '''

     :

    N = len(state)
