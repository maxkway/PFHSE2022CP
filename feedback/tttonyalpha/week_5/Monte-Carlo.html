def getSample(N):
    """
           ~sqrt(1-x^2) 

    --------
    :
    N - int,   
    --------
     :
    sample - numpy ndarray of float,     N
    """
    sample = []
    cnt = 0
    while cnt < N:
        x=np.random.uniform(-1,1)
        y=np.random.uniform(0,1)
        if y < np.sqrt(1 - x ** 2):
            sample.append(x)
            cnt+=1

    return sample
