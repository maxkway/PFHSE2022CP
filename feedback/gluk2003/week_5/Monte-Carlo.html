def getSphereVolumePseudorandom(N: int, D: int, R = 1):
    """
        D-   R  -   

    --------
    :
    N - int,   
    D - int,   
    R = 1 - float,  
    --------
     :
    V - float,  
    """
    def inside(v, r):
        return (v ** 2).sum() < r ** 2
    
    
    set_of_points = np.random.uniform(low = -R, high = R, size = (N, D))
    N_iside = np.count_nonzero([inside(set_of_points[i], R) for i in range(len(set_of_points))])
    Omega_C = (2 * R) ** D
    V = N_iside / N * Omega_C

    return V

def getSample(N):
    """
           ~sqrt(1-x^2) 

    --------
    :
    N - int,   
    --------
     :
    sample - numpy ndarray of float,     N
    """
    np.random.seed(42)
    w = lambda x: np.sqrt(1 - x ** 2)
    
    sample = []
    while len(sample) < N:
      x = np.random.uniform(-1, 1, size=1)
      y = np.random.uniform(0, 1, size=1)
      if y[0] < w(x[0]):  
        sample.append(x[0])

    return sample

def getEnergy(state):
    '''
    ,     
    ---------
    :
    state - numpy ndarray of ints,     NxN
    --------
     :
    E - float,   
    '''
    N = len(state)
    E = 0
    for i in range(N):
      for j in range(N):
        E_ij = state[i][j] * (state[(i - 1) % N][j] + state[(i + 1) % N][j] + state[i][(j - 1) % N] + state[i][(j + 1) % N])
        E += E_ij
    
    return float(E)
