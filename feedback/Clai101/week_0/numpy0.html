def make_curve(n, a):
    '''   n2,  n  
        '''
    phi = np.linspace(0, 2 * np.pi, n)
    r = a + np.sin(phi)
    xy = np.array([[r[i] * np.cos(phi[i]), r[i] * np.sin(phi[i])] for i in range(n)])
    return xy
