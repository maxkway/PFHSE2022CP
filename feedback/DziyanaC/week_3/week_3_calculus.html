def deriv(f, x, h):
    """   `f`   `x`   `h`.
     ,        $O(h^2)$.
    
    Parameters
    ----------
    f : callable
        ,   
    x : float
        ,     
    h : float
        
        
    Rerurns
    -------
    fder :  f(x)   x   h.
    """
    u_i = f(x)
    u_ii = f(x + h)
    u_iii = f(x + 2 * h)
    fder = (-3 * u_i + 4 * u_ii - u_iii) / (2 * h)
    return fder

def midpoint_rule(func, a, b, eps):
    """   f  a  b    .
    
    Parameters
    ----------
    func : callable
        ,   
    a : float
          
    b : float
          
    eps : float
          
        
    Returns
    -------
    integral : float
          $\int_a^b f(x) dx$.
    n : int
        The number of intervals on segment.
    """
    N = 1
    midp = (a + b) / 2
    int = b - a
    int_n = func(midp) * int
    int_2n = (func((a + midp) / 2) + func((midp + b) / 2)) * int / 2
    while abs(int_2n - int_n) >= eps:
        N += 1

        h1 = int / N
        i = np.arange(1, N + 1)
        int_n = (func(a + (i - 0.5) * h1) * h1).sum()

        h2 = int / (2 * N)
        i = np.arange(1, 2 * N + 1)
        int_2n = (func(a + (i - 0.5) * h2) * h2).sum()

    integral = int_n 
    return integral, N
