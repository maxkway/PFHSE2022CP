def lhs_matrix(n):
    #  
    a = np.zeros((n, n))

    idx = np.arange(n)

    a[idx, idx] = -4 # u_{ij}
    a[idx[:-1], idx[:-1]+1] = 1 #   x
    a[idx[1:], idx[1:]-1] = 1

    #  -  `m`
    m = block_diag(*(a,)*n)

    #  "" -   y
    idx = np.arange(m.shape[0])

    m[idx[:-n], idx[:-n] + n] = 1
    m[idx[n:], idx[n:] - n] = 1
    return m

def seidel_solve(m, b, x0, eps=1e-5, maxiter=500):
   n = len(m)
   x = np.zeros(n)

   Iteration = 0
   converge = True
   pogr = 0
   while converge and Iteration < maxiter:
      x_new = np.copy(x)
      for i in range(n):
         s1 = sum(m[i][j] * x_new[j] for j in range(i))
         s2 = sum(m[i][j] * x[j] for j in range(i + 1, n))
         x_new[i] = (b[i] - s1 - s2) / m[i][i]
      pogr = max(abs(x_new[i] - x[i])  for i in range(n))
      converge =  np.linalg.norm(x_new - x) > eps
      Iteration += 1
      x = x_new
   return x, Iteration
