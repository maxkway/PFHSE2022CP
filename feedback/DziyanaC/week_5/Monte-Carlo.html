def getSphereVolumePseudorandom(N, D, R = 1):
    """
        D-   R  -   

    --------
    :
    N - int,   
    D - int,   
    R = 1 - float,  
    --------
     :
    V - float,  
    """
    p = 0
    random = np.random.RandomState(12345)
    x = random.uniform(low=-1, high=1, size=(D, N))
    for i in range(D):
        p += x[i]**2
    m = p < R**2
    V = 2**D * np.count_nonzero(m) / N
    return V

def getSample(N):
    """
           ~sqrt(1-x^2) 

    --------
    :
    N - int,   
    --------
     :
    sample - numpy ndarray of float,     N
    """
    w = lambda x: np.sqrt(1 - x ** 2)
    random = np.random.RandomState(12345)
    sample = []
    while len(sample) < N:
        x = random.uniform(low=-1, high=1, size=1)
        y = random.uniform(low=0, high=1, size=1)
        if y[0] < w(x[0]):
            sample.append(x[0])
    return sample

    '''
           (i,j)     (   state)

    dE - float,

def makeFlip(T, state):
    '''
     N^2   ,        ,   4.  5.

    N = len(state)

        i = np.random.randint(N, size=1)[0]

        dE = getDeltaE(i, j, state)
        if dE < 0:
            state[i][j] = -state[i][j]
