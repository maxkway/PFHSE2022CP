import scipy
def gauss_2(a, b):
    r"""Return nodes and weights for a two-point Gauss quadrature on [a, b].
    
    Parameters
    ----------
    a, b : float
       Boundaries of the integration interval
       
    Returns
    -------
    x1, x2, w1, w2 : float
       Nodes and weights of the quadrature.
    """
    x, w = scipy.special.roots_legendre(2)
    w1 = (b-a)*w[0]/2
    w2 = (b-a)*w[1]/2
    x1 = (a*(1-x[0]) + b*(1 + x[0]))/2
    x2 = (a*(1-x[1]) + b*(1+x[1]))/2
    return x1, x2, w1, w2

import numpy as np

def midpoint_rule(func, a, b, eps):
    
    I = float('inf')
    i = 0
    N = 2
    
    while i == 0:
        
        hk = (b-a)/N
        xk = np.linspace(a + hk/2, b - hk/2, N)
        Qk = [func(k) for k in xk]
        Q = np.sum(Qk)*hk
        
        if (abs(I-Q)<eps):
            break
            
        N *= 2
        I = Q
        
    return Q, N

func = lambda x: 7*x**3 - 8*x**2 - 3*x + 3
I, N = midpoint_rule(func, -1, 1, 1e-10)
print("I =", I, ", The number of nodes:", N+1)
