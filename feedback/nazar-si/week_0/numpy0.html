def make_curve(n, a):
    '''   n2,  n  
        '''
    t = np.linspace(0, 2 * np.pi, n)
    xy = np.array([a * np.cos(t) - np.cos(2 * t), a * np.sin(t) - np.sin(2 * t)]).T
    return xy
