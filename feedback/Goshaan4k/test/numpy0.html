def make_curve(n, a):
    '''   n2,  n  
        '''
    phi = np.linspace(1, 10, num=n)
    r = a + np.cos(phi)
    xy = np.empty((n, 2))
    xy[:, 0] = r * np.cos(phi)
    xy[:, 1] = r * np.sin(phi)
    return xy
