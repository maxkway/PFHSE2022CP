def deriv(f, x, h):
    """   `f`   `x`   `h`.
     ,        $O(h^2)$.
    
    Parameters
    ----------
    f : callable
        ,   
    x : float
        ,     
    h : float
        
        
    Rerurns
    -------
    fder :  f(x)   x   h.
    """
    fder=(f(x+h)-f(x-h))/(2*h)
    return fder

def midpoint_rule(func, a, b, eps):
    """   f  a  b    .
    
    Parameters
    ----------
    func : callable
        ,   
    a : float
          
    b : float
          
    eps : float
          
        
    Returns
    -------
    integral : float
          $\int_a^b f(x) dx$.
    n : int
        The number of intervals on segment.
    """
    n=2
    h=(b-a)/n
    I_n=(b-a)*func((a+b)/2)
    I_n1=(func(a+0.5*h)+func(a+1.5*h))*h
    while abs(I_n1-I_n)>eps: #   
        I_n=I_n1
        n=n*2
        h=(b-a)/n
        Ik_list=[]
        for n in range(1,n):
            x1=a + (n-1)*h
            x2=a + n*h
            x=(x1+x2)/2
            Ik=func(x)*h
            Ik_list.append(Ik)
        I_n1=sum(Ik_list)
    integral=I_n
    return integral, n
