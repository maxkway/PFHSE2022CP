def deriv(f, x, h):
    """   `f`   `x`   `h`.
     ,        $O(h^2)$.
    
    Parameters
    ----------
    f : callable
        ,   
    x : float
        ,     
    h : float
        
        
    Rerurns
    -------
    fder :  f(x)   x   h.
    """
    # YOUR CODE HERE
    function = f(x)
    f1 = f(x + h)
    f2 = f(x + 2 * h)
    fder = (-3*function+4*f1-f2)/(2*h)
    return fder

    chis = f(x + h) - f(x)
    fder = chis/ h

def der_three(x, h):
    a = np.array([[1, 1, 1], [0, 1, 2], [0, 1, 4]])
    b = np.array([0, 1, 0])
    c = np.linalg.solve(a, b)
    chis= c[0] * f(x) + c[1] * f(x + h) + c[2] * f(x + 2 * h)
    fder = chis/ h
    return fder

import numpy as np
import matplotlib.pyplot as plt

def midpoint_rule(func, a, b, eps):
    n = 1
    I_n = func((a + b) / 2) * (b - a)
    I_2n = (func((a + (a + b) / 2) / 2) + func(((a + b) / 2 + b) / 2)) * (b - a) / 2
    while abs(I_2n - I_n) >= eps:
        n += 1
        h = (b - a) / n
        i = np.arange(1, n + 1)
        I_n = (func((i - 0.5) * h) * h).sum()
        
        h = (b - a) / (2 * n)
        i = np.arange(1, 2 * n + 1)
        I_2n = (func((i - 0.5) * h) * h).sum()

    integral = I_n 
    return integral, n
