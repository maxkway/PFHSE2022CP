def MonteCarloNormal(M1, dM1, M2, dM2, r, dr, n=10**6):
    G = 6.67384 * 10 ** (-11)
    r_ar = np.random.normal(r, dr, (1, n))
    M1_ar = np.random.normal(M1, dM1, (1, n))
    M2_ar = np.random.normal(M2, dM2, (1, n))
    F = (G * M1_ar * M2_ar )/ r_ar ** 2
    return F[0]
