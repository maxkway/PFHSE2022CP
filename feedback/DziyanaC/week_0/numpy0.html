       r phi.
    
    Parameters

    c: Array of int.

    -------
    p: Array of int. 
         .    r.
    """
    x = c[:, 0]
    y = c[:, 1]
    r = np.sqrt(x**2 + y**2)
    phi = np.arctan2(y, x)

def make_curve(n, a):
    '''   n2,  n  
        '''
    
    phi = np.linspace(0, 2 * np.pi, n)
    r = np.cos(phi) + a
    
    def cartesian_to_polar(r, phi):
        x = r * np.cos(phi)
        y = r * np.sin(phi)
        return x, y
    return cartesian_to_polar(r,phi)
