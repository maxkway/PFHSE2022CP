def newton_iteration(f, fder, x0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    f : callable
        ,     .
    fder : callable
         `f`.
    x0 : float
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    x : float
           .
    niter : int
         .
    """
    x_nplus1 = x0
    x_n = x0 + 2 * eps
    i = 0

    while abs(x_nplus1 - x_n) >= eps and i < maxiter:
        x_n = x_nplus1
        x_nplus1 = x_n - f(x_n) / fder(x_n)
        i += 1
    
    x = float(x_nplus1)
    n_iter = i
    return x, n_iter

The Newton's iterations converge to one of these solutions, depending on the starting point in the complex plane (to converge to a complex-valued solution, the iteration needs a complex-valued starting point).

def newton_iteration_complex(f, fder, x0, eps=1e-5, maxiter=100):
    x_nplus1 = x0
    x_n = x0 + 2 * complex(eps, eps)
    i = 0

    while abs(x_nplus1 - x_n) >= eps and i < maxiter:
        x_n = x_nplus1
        x_nplus1 = x_n - f(x_n) / fder(x_n)
        i += 1

def newton_system(F, jcb, X0, eps=1e-5, maxiter=100):
    """  $f(x) = 0$   .
    
    Parameters
    ----------
    F : callable
        - ,    .
    jcb : callable
         `f`.
    X0 : array-like of floats, shape (n,)
           .
    eps : float
         . 
                 `eps`.
          1e-5.
    maxiter : int
           (  100).
         ,     `maxiter`.
               .
    
    Returns
    -------
    X : array-like of floats, shape (n,)
           .
    niter : int
         .
    """
    x_nplus1 = X0
    x_n = X0 + 2 * eps
    i = 0

    while np.linalg.norm(x_nplus1 - x_n) >= eps and i < maxiter:
        x_n = x_nplus1
        x_nplus1 = x_n + scipy.linalg.solve(jcb(x_n), -F(x_n))
        i += 1
    X = x_nplus1
    return X, i

def func2(X):
    x, y, z = X
    return np.array([x**2 + y**2 + z**2 - 25,
                     x*y + y*z + z*x - 5,
                     x + y - 3])

def jac2(X):
    r"""Return the Jacobian of `func2(X)`.
    
    
    The Jacobian matrix is defined as 
    
    $$
    J_{ij} = \partial f_i / \partial x_j
    $$
    
    so that the first row contains the derivatives of $f_0$
    with respect to the first, second etc coordinates; the second
    row contains the derivatives of $f_1$ with respect to
    the first, second etc coordinates; and so on.
    """
    x1, y1, z1 = X
    x, y, z = sympy.symbols('x y z')
    f1, f2, f3 = func2([x, y, z])
    F = sympy.Matrix([f1, f2, f3])
    J = F.jacobian([x, y ,z])
    s = (x, y, z)
    J_func = sympy.lambdify(s, J, modules='numpy')
    J_ev = J_func(x1, y1, z1)
    
    return J_ev
