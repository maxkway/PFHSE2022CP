def midpoint_rule(func, a, b, eps):
    """   f  a  b    .
    
    Parameters
    ----------
    func : callable
        ,   
    a : float
          
    b : float
          
    eps : float
          
        
    Returns
    -------
    integral : float
          $\int_a^b f(x) dx$.
    n : int
        The number of intervals on segment.
    """
    def i(f, a, b, n):
        d = (b - a) / n
        return np.sum(np.array([d * f(a + (j + 0.5) * d) for j in range(n)]))

    n = int((b - a) / eps ** 0.5)
    i1 = i(func,a,b,n)
    while True: 
        i2 = i(func,a,b,2 * n)
        if abs(i2 - i1) < eps:
            return i2, n * 2 
        i1 = i2 
        n *= 2
