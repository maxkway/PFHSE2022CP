def lhs_matrix(n):
    #  
    a = np.zeros((n, n))

    idx = np.arange(n)

    a[idx, idx] = -4 # u_{ij}
    a[idx[:-1], idx[:-1]+1] = 1 #   x
    a[idx[1:], idx[1:]-1] = 1

    #  -  `m`
    m = block_diag(*(a,)*n)

    #  "" -   y
    idx = np.arange(m.shape[0])

    m[idx[:-n], idx[:-n] + n] = 1
    m[idx[n:], idx[n:] - n] = 1
    return m

from numpy import linalg as g
def seidel_solve(m, b, x0, eps=1e-5, maxiter=500):
    """ Solve `m @ x = b` with Seidel iteration.
    
    Parameters
    ----------
    m : ndarray, shape (n, n)
       Input matrix
    b : ndarray, shape (n,)
       Right-hand side
    x0 : ndarray, shape (n, )
       Initial point for the iteration
    eps : float, optional
       Target accuracy.
       Stop iterations when the 2-norm of 
       `|x_k - x_{k-1}| < eps`
    maxiter : int
       Maximum allowed number of iterations.
       
    Returns
    -------
    x : ndarray, shape (n,)
       The approximation to the solution of `m @ x = b`
    nit : the number of iterations performed.
    """
    n = len(m)
    x = x0
    nit = 0
    p = True
    while p and nit < maxiter:
        x1 = np.copy(x)
        for i in range(n):
            a = sum(m[i][j] * x1[j] for j in range(i)) + sum(m[i][k] * x1[k] for k in range(i + 1, n))
            x1[i] = (b[i] - a) / m[i][i]
        p = g.norm(x1 - x, 2) >= eps
        x = x1
        nit += 1
    return x, nit
